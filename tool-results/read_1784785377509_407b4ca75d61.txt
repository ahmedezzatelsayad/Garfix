    20тЖТ    20тЖТfunction countFiles(dir: string, pattern: RegExp, excludePattern?: RegExp): number {
    21тЖТ    21тЖТ  let count = 0;
    22тЖТ    22тЖТ  function walk(d: string) {
    23тЖТ    23тЖТ    try {
    24тЖТ    24тЖТ      for (const entry of readdirSync(d)) {
    25тЖТ    25тЖТ        const full = join(d, entry);
    26тЖТ    26тЖТ        const st = statSync(full);
    27тЖТ    27тЖТ        if (st.isDirectory()) {
    28тЖТ    28тЖТ          if (!excludePattern || !excludePattern.test(full)) walk(full);
    29тЖТ    29тЖТ        } else if (pattern.test(entry)) {
    30тЖТ    30тЖТ          if (!excludePattern || !excludePattern.test(full)) count++;
    31тЖТ    31тЖТ        }
    32тЖТ    32тЖТ      }
    33тЖТ    33тЖТ    } catch { /* skip */ }
    34тЖТ    34тЖТ  }
    35тЖТ    35тЖТ  walk(dir);
    36тЖТ    36тЖТ  return count;
    37тЖТ    37тЖТ}
