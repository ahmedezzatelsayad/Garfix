    31→/* ─── Types ────────────────────────────────────────────────────────────────── */
    32→type ModuleTab =
    33→  | "core"          // existing monolithic content (accounts, journal, trial, statements, fiscal-periods, cost-centers, aging, banking, dashboard)
    34→  | "dashboard"     // Financial dashboard (calls /api/accounting/financial-dashboard)

    49→type StatementType = "profit-loss" | "balance-sheet" | "cash-flow";

    53→interface JournalEntry { id: number; date: string; description?: string; reference?: string; status: string; lines: JournalLine[]; }

    55→  id: number; code: string; nameAr: string; type: string;
    56→  totalDebit: number; totalCredit: number; balance: number;
    57→}
    58→interface FiscalPeriod { id: number; name: string; startDate: string; endDate: string; status: string; closedAt?: string; }
    59→interface CostCenter { id: number; code: string; nameAr: string; parentId?: number | null; type: string; budget?: number; actual?: number; }
    60→interface AgingBucket { range: string; receivable: number; payable: number; count: number; }
    61→interface BankAccount { id: number; name: string; bankName: string; accountNumber: string; currency: string; balance: number; iban?: string; }

    83→export function AccountingView() {
    84→  const { activeCompany } = useBrand();
    85→  const [moduleTab, setModuleTab] = useState<ModuleTab>("core");
    86→
    87→  /* ─── Core sub-module state (preserved exactly as-is) ────────────────── */
    88→  const [tab, setTab] = useState<Tab>("accounts");
    89→  const [accounts, setAccounts] = useState<Account[]>([]);
    90→  const [entries, setEntries] = useState<JournalEntry[]>([]);
    91→  const [trial, setTrial] = useState<{ accounts: TrialRow[]; grandDebit: number; grandCredit: number; isBalanced: boolean } | null>(null);
    92→  const [fiscalPeriods, setFiscalPeriods] = useState<FiscalPeriod[]>([]);
    93→  const [costCenters, setCostCenters] = useState<CostCenter[]>([]);
    94→  const [agingData, setAgingData] = useState<AgingBucket[]>([]);
    95→  const [bankAccounts, setBankAccounts] = useState<BankAccount[]>([]);
    96→  const [loading, setLoading] = useState(true);
    97→  const [showForm, setShowForm] = useState(false);
    98→  const [selectedIds, setSelectedIds] = useState<Set<number>>(new Set());
    99→  const [currentPage, setCurrentPage] = useState(1);
   100→  const [bulkDeleting, setBulkDeleting] = useState(false);
   101→  const [reversingId, setReversingId] = useState<number | null>(null);
   102→  const [reverseConfirm, setReverseConfirm] = useState<JournalEntry | null>(null);
   103→
   104→  const load = useCallback(async () => {
   105→    if (!activeCompany) { setLoading(false); return; }
   106→    setLoading(true);
   107→    const slug = `companySlug=${encodeURIComponent(activeCompany.slug)}`;
   108→    try {
   109→      const [a, j] = await Promise.all([
   110→        authedFetch(`/api/accounting/accounts?${slug}`),
   111→        authedFetch(`/api/accounting/journal-entries?${slug}`),
   112→      ]);
   113→      if (!a.ok || !j.ok) {
   114→        const failed = !a.ok ? a : j;
   115→        const e = await failed.json().catch(() => ({}));
   116→        throw new Error((e as Record<string, unknown>)?.error as string || `فشل تحميل البيانات (${failed.status})`);
   117→      }
   118→      const [aD, jD] = await Promise.all([a.json(), j.json()]);
   119→      setAccounts((aD as { accounts?: Account[] }).accounts || []);
   120→      setEntries((jD as { entries?: JournalEntry[] }).entries || []);
   121→    } catch (err) {
   122→      toast.error(err instanceof Error ? err.message : "تعذّر تحميل بيانات المحاسبة");
   123→    } finally { setLoading(false); }
   124→  }, [activeCompany]);
   125→
   126→  const loadTrial = useCallback(async () => {
   127→    if (!activeCompany) return;
   128→    try {
   129→      const res = await authedFetch(`/api/accounting/trial-balance?companySlug=${encodeURIComponent(activeCompany.slug)}`);
   130→      if (!res.ok) {
   131→        const e = await res.json().catch(() => ({}));
   132→        toast.error((e as Record<string, unknown>)?.error as string || "تعذّر تحميل ميزان المراجعة");
   133→        setTrial(null);
   134→        return;
   135→      }
   136→      setTrial(await res.json());
   137→    } catch (err) {
   138→      toast.error(err instanceof Error ? err.message : "تعذّر تحميل ميزان المراجعة");
   139→      setTrial(null);
   140→    }
   141→  }, [activeCompany]);
   142→
   143→  const loadFiscalPeriods = useCallback(async () => {
   144→    if (!activeCompany) return;
   145→    try {
   146→      const res = await authedFetch(`/api/accounting/fiscal-periods?companySlug=${encodeURIComponent(activeCompany.slug)}`);
   147→      if (res.ok) { const d = await res.json(); setFiscalPeriods(d.periods || []); }
   148→      else { setFiscalPeriods([]); }
   149→    } catch { setFiscalPeriods([]); }
   150→  }, [activeCompany]);
   151→
   152→  const loadCostCenters = useCallback(async () => {
   153→    if (!activeCompany) return;
   154→    try {
   155→      const res = await authedFetch(`/api/accounting/cost-centers?companySlug=${encodeURIComponent(activeCompany.slug)}`);
   156→      if (res.ok) { const d = await res.json(); setCostCenters(d.costCenters || []); }
   157→      else { setCostCenters([]); }
   158→    } catch { setCostCenters([]); }
   159→  }, [activeCompany]);
   160→
   161→  const loadAging = useCallback(async () => {
   162→    if (!activeCompany) return;
   163→    try {
   164→      const res = await authedFetch(`/api/accounting/aging?companySlug=${encodeURIComponent(activeCompany.slug)}`);
   165→      if (res.ok) { const d = await res.json(); setAgingData(d.buckets || []); }
   166→      else { setAgingData([]); }
   167→    } catch { setAgingData([]); }
   168→  }, [activeCompany]);
   169→
   170→  const loadBankAccounts = useCallback(async () => {
   171→    if (!activeCompany) return;
   172→    try {
   173→      const res = await authedFetch(`/api/accounting/bank-accounts?companySlug=${encodeURIComponent(activeCompany.slug)}`);
   174→      if (res.ok) { const d = await res.json(); setBankAccounts(d.accounts || []); }
   175→      else { setBankAccounts([]); }
   176→    } catch { setBankAccounts([]); }
   177→  }, [activeCompany]);
   178→
   179→  useEffect(() => { load(); }, [load]);
   180→  useEffect(() => {
   181→    if (tab === "trial" && activeCompany) loadTrial();
   182→    if (tab === "fiscal-periods" && activeCompany) loadFiscalPeriods();
   183→    if (tab === "cost-centers" && activeCompany) loadCostCenters();
   184→    if (tab === "aging" && activeCompany) loadAging();
   185→    if (tab === "banking" && activeCompany) loadBankAccounts();
   186→    if (tab === "dashboard" && activeCompany) { load(); loadAging(); loadBankAccounts(); }
   187→  }, [tab, activeCompany, loadTrial, loadFiscalPeriods, loadCostCenters, loadAging, loadBankAccounts, load]);
   188→
   189→  const switchTab = (t: Tab) => {
   190→    setTab(t);
   191→    setShowForm(false);
   192→    setSelectedIds(new Set());
   193→    setCurrentPage(1);
   194→  };
   195→
   196→  const itemsForTab = (): Array<{ id: number }> => (tab === "accounts" ? accounts : entries);
   197→
   198→  const allItems = itemsForTab();
   199→  const totalPages = Math.max(1, Math.ceil(allItems.length / PAGE_SIZE));
   200→  const pageItems = allItems.slice((currentPage - 1) * PAGE_SIZE, currentPage * PAGE_SIZE);
   201→  const safePage = Math.min(currentPage, totalPages);
   202→
   203→  const toggleSelectAll = () => {
   204→    if (selectedIds.size === pageItems.length && pageItems.length > 0) setSelectedIds(new Set());
   205→    else setSelectedIds(new Set(pageItems.map((i) => i.id)));
   206→  };
   207→  const toggleRow = (id: number) => {
   208→    setSelectedIds((prev) => {
   209→      const next = new Set(prev);
   210→      if (next.has(id)) next.delete(id);
   211→      else next.add(id);
   212→      return next;
   213→    });
   214→  };
   215→  const handleBulkDelete = async () => {
   216→    if (selectedIds.size === 0) return;
   217→    if (!confirm(`حذف ${selectedIds.size} عنصر؟`)) return;
   218→    const endpoint = tab === "accounts" ? "/api/accounting/accounts" : "/api/accounting/journal-entries";
   219→    setBulkDeleting(true);
   220→    let okCount = 0, failCount = 0;
   221→    for (const id of selectedIds) {
   222→      try {
   223→        const res = await authedFetch(`${endpoint}/${id}`, { method: "DELETE" });
   224→        if (res.ok) okCount++;
   225→        else { failCount++; }
   226→      } catch { failCount++; }
   227→    }
   228→    setBulkDeleting(false);
   229→    setSelectedIds(new Set());
   230→    if (okCount > 0) toast.success(`تم حذف ${okCount} عنصر`);
   231→    if (failCount > 0) toast.error(`تعذّر حذف ${failCount} عنصر`);
   232→    load();
   233→  };
   234→  const handleDelete = async (id: number) => {
   235→    if (!confirm("حذف هذا العنصر؟")) return;
   236→    const endpoint = tab === "accounts" ? "/api/accounting/accounts" : "/api/accounting/journal-entries";
   237→    const res = await authedFetch(`${endpoint}/${id}`, { method: "DELETE" });
   238→    if (res.ok) { toast.success("تم الحذف"); load(); }
   239→    else { const e = await res.json().catch(() => ({})); toast.error(e.error || "تعذّر الحذف"); }
   240→  };
   241→
   242→  const handleReverse = async (entry: JournalEntry) => {
   243→    if (!activeCompany) return;
   244→    setReverseConfirm(entry);
   245→  };
   246→
   247→  const confirmReverse = async () => {
   248→    if (!reverseConfirm || !activeCompany) return;
   249→    const entry = reverseConfirm;
   250→    setReversingId(entry.id);
   251→    try {
   252→      const url = `/api/accounting/journal-entries/${entry.id}/reverse?companySlug=${encodeURIComponent(activeCompany.slug)}`;
   253→      const res = await authedFetch(url, { method: "POST" });
   254→      if (!res.ok) {
   255→        const e = await res.json().catch(() => ({}));
   256→        throw new Error(e.error || "تعذّر العكس");
   257→      }
   258→      const data = await res.json().catch(() => ({}));
   259→      const reversalId = data?.reversal?.id;
   260→      toast.success(reversalId ? `تم إنشاء قيد عكسي #${reversalId}` : "تم عكس القيد");
   261→      setReverseConfirm(null);
   262→      await load();
   263→    } catch (err) {
   264→      toast.error(err instanceof Error ? err.message : "خطأ أثناء العكس");
   265→    } finally { setReversingId(null); }
   266→  };
   267→
   268→  const pageBtnStyle = (disabled: boolean): string =>
   269→    disabled
   270→      ? "py-1.5 px-3 rounded-md bg-transparent text-muted-foreground border border-border text-[12px] font-bold cursor-not-allowed opacity-50"
   271→      : "py-1.5 px-3 rounded-md bg-card text-foreground border border-border text-[12px] font-bold cursor-pointer";
   272→
   273→  if (!activeCompany) return <div className="p-12 text-center text-muted-foreground">اختر شركة</div>;
   274→
   275→  const ACCOUNT_TYPES: Record<string, { label: string; color: string }> = {
   276→    asset: { label: "أصول", color: "#10b981" },
   277→    liability: { label: "خصوم", color: "#ef4444" },
   278→    equity: { label: "حقوق ملكية", color: "#7c3aed" },
   279→    revenue: { label: "إيرادات", color: "#3b82f6" },
   280→    expense: { label: "مصروفات", color: "#f59e0b" },
   281→    contra_revenue: { label: "مقابل إيرادات", color: "#9ca3af" },
   282→    contra_asset: { label: "مقابل أصول", color: "#9ca3af" },
   283→  };
   284→
   285→  const tabs: Array<{ key: Tab; label: string; icon?: React.ComponentType<{ size?: number }> }> = [
   286→    { key: "dashboard", label: "لوحة التحكم", icon: LayoutDashboard },
   287→    { key: "accounts", label: `الحسابات (${accounts.length})` },
   288→    { key: "journal", label: `القيود (${entries.length})` },
   289→    { key: "trial", label: "ميزان المراجعة", icon: Scale },
   290→    { key: "statements", label: "القوائم المالية", icon: FileBarChart },
   291→    { key: "fiscal-periods", label: "الفترات المالية", icon: Clock },
   292→    { key: "cost-centers", label: "مراكز التكلفة", icon: Building },
   293→    { key: "aging", label: "تقادم الذمم", icon: BarChart3 },
   294→    { key: "banking", label: "البنوك", icon: Landmark },
   295→  ];
   296→
   297→  // Dashboard metrics
   298→  const totalRevenue = accounts.filter(a => a.type === "revenue").reduce((s, a) => s + a.balance, 0);
   299→  const totalExpenses = accounts.filter(a => a.type === "expense").reduce((s, a) => s + a.balance, 0);
   300→  const totalAssets = accounts.filter(a => a.type === "asset").reduce((s, a) => s + a.balance, 0);
   301→  const totalLiabilities = accounts.filter(a => a.type === "liability").reduce((s, a) => s + a.balance, 0);
   302→  const netProfit = totalRevenue - totalExpenses;
   303→  const totalAR = agingData.reduce((s, b) => s + b.receivable, 0);
   304→  const totalAP = agingData.reduce((s, b) => s + b.payable, 0);
   305→  const totalCash = bankAccounts.reduce((s, b) => s + b.balance, 0);
   306→  const currentPeriod = fiscalPeriods.find(p => p.status === "open");
   307→
   308→  /* ─── Render ──────────────────────────────────────────────────────────── */
   309→  return (
   310→    <div className="flex flex-col gap-4">
   311→      {/* Module-level header */}
   312→      <div className="flex flex-wrap justify-between items-center gap-3">
   313→        <div><h1 className="text-2xl font-extrabold flex items-center gap-2"><Calculator size={20} /> المحاسبة</h1><p className="text-[13px] text-muted-foreground">{activeCompany.nameAr || activeCompany.name}</p></div>
   314→      </div>
   315→
   316→      {/* Module-level navigation tabs (top row, visually distinct) */}
   317→      <div className="flex gap-1.5 flex-wrap bg-muted/50 rounded-[12px] p-2">
   318→        {MODULE_TABS.map((mt) => {
   319→          const Icon = mt.icon;
   320→          const active = moduleTab === mt.key;
   321→          return (
   322→            <button
   323→              key={mt.key}
   324→              onClick={() => setModuleTab(mt.key)}
   325→              className={cn(
   326→                "py-2.5 px-4 rounded-[10px] text-[13px] font-bold cursor-pointer inline-flex items-center gap-2 transition-colors",
   327→                active
   328→                  ? "bg-primary text-primary-foreground shadow-sm"
   329→                  : "bg-card text-muted-foreground border border-border hover:bg-accent hover:text-accent-foreground",
   330→              )}
   331→            >
   332→              <Icon size={16} />
   333→              {mt.label}
   334→            </button>
   335→          );
   336→        })}
   337→      </div>
   338→
   339→      {/* ─── Module content ─────────────────────────────────────────────── */}
   340→      {moduleTab === "core" ? (
   341→        /* ─── Core sub-module: exact original inner content ───────────── */
   342→        <>
   343→          <div className="flex flex-wrap justify-between items-center gap-3">
   344→            {(tab === "accounts" || tab === "journal" || tab === "fiscal-periods" || tab === "cost-centers" || tab === "banking") && (
   345→              <button onClick={() => setShowForm(true)} className="inline-flex items-center gap-1.5 py-2.5 px-[18px] rounded-[10px] bg-primary text-primary-foreground border-none text-[13px] font-bold cursor-pointer"><Plus size={16} /> إضافة</button>
   346→            )}
   347→          </div>
   348→          <div className="flex gap-1.5 flex-wrap">
   349→            {tabs.map((t) => {
   350→              const Icon = t.icon;
   351→              return (
   352→                <button key={t.key} onClick={() => switchTab(t.key)} className={cn(
   353→                  "py-2 px-4 rounded-[10px] border border-border text-[12px] font-bold cursor-pointer inline-flex items-center gap-1.5",
   354→                  tab === t.key ? "bg-primary text-primary-foreground" : "bg-card text-muted-foreground",
   355→                )}>
   356→                  {Icon && <Icon size={14} />}
   357→                  {t.label}
   358→                </button>
   359→              );
   360→            })}
   361→          </div>
   362→
   363→          {loading ? <div className="p-12 text-center text-muted-foreground">جارٍ التحميل…</div> : tab === "dashboard" ? (
   364→            <FinancialDashboard
   365→              totalRevenue={totalRevenue}
   366→              totalExpenses={totalExpenses}
   367→              netProfit={netProfit}
   368→              totalAssets={totalAssets}
   369→              totalLiabilities={totalLiabilities}
   370→              totalAR={totalAR}
   371→              totalAP={totalAP}
   372→              totalCash={totalCash}
   373→              currentPeriod={currentPeriod}
   374→              entriesCount={entries.length}
   375→              accountsCount={accounts.length}
   376→            />
   377→          ) : tab === "trial" ? (
   378→            <TrialBalanceTable data={trial} loading={loading} />
   379→          ) : tab === "statements" ? (
   380→            <FinancialStatements company={activeCompany} />
   381→          ) : tab === "fiscal-periods" ? (
   382→            showForm ? <FiscalPeriodForm company={activeCompany} onClose={() => setShowForm(false)} onSaved={() => { setShowForm(false); loadFiscalPeriods(); }} /> : (
   383→              <FiscalPeriodsTable periods={fiscalPeriods} company={activeCompany} onRefresh={loadFiscalPeriods} />
   384→            )
   385→          ) : tab === "cost-centers" ? (
   386→            showForm ? <CostCenterForm company={activeCompany} costCenters={costCenters} onClose={() => setShowForm(false)} onSaved={() => { setShowForm(false); loadCostCenters(); }} /> : (
   387→              <CostCentersTable costCenters={costCenters} />
   388→            )
   389→          ) : tab === "aging" ? (
   390→            <AgingReport data={agingData} totalAR={totalAR} totalAP={totalAP} />
   391→          ) : tab === "banking" ? (
   392→            showForm ? <BankAccountForm company={activeCompany} onClose={() => setShowForm(false)} onSaved={() => { setShowForm(false); loadBankAccounts(); }} /> : (

  1402→function JournalForm({ company, accounts, onClose, onSaved }: { company: { slug: string }; accounts: Account[]; onClose: () => void; onSaved: () => void }) {
  1403→  const [date, setDate] = useState(new Date().toISOString().slice(0, 10));
  1404→  const [description, setDescription] = useState("");
  1405→  const [reference, setReference] = useState("");
  1406→  const [status, setStatus] = useState("draft");
  1407→  const [lines, setLines] = useState<Array<{ accountId: number | null; debit: number; credit: number; description?: string }>>([{ accountId: null, debit: 0, credit: 0 }]);
  1408→  const [saving, setSaving] = useState(false);
  1409→
  1410→  const totalDebit = lines.reduce((s, l) => s + Number(l.debit || 0), 0);
  1411→  const totalCredit = lines.reduce((s, l) => s + Number(l.credit || 0), 0);
  1412→  const isBalanced = Math.abs(totalDebit - totalCredit) < 0.001;
  1413→
  1414→  const updateLine = (i: number, field: string, value: number | string) => { setLines((arr) => arr.map((l, idx) => idx === i ? { ...l, [field]: value } : l)); };
  1415→  const addLine = () => setLines((arr) => [...arr, { accountId: null, debit: 0, credit: 0 }]);
  1416→  const removeLine = (i: number) => setLines((arr) => arr.filter((_, idx) => idx !== i));
  1417→
  1418→  const submit = async () => {
  1419→    if (lines.length === 0) { toast.error("أضف بنداً واحداً على الأقل"); return; }
  1420→    if (!isBalanced) { toast.error("القيد غير متوازن — المدين ≠ الدائن"); return; }
  1421→    if (lines.some((l) => !l.accountId)) { toast.error("كل بند يجب أن يحدد حساباً"); return; }
  1422→    setSaving(true);
  1423→    try {
  1424→      const res = await authedFetch("/api/accounting/journal-entries", {
  1425→        method: "POST", headers: { "Content-Type": "application/json" },
  1426→        body: JSON.stringify({ date, description, reference, status, lines, companySlug: company.slug }),
  1427→      });
  1428→      if (!res.ok) { const e = await res.json().catch(() => ({})); throw new Error(e.error || "Failed"); }
  1429→      toast.success("تم إنشاء القيد"); onSaved();
  1430→    } catch (err) { toast.error(err instanceof Error ? err.message : "خطأ"); }
  1431→    finally { setSaving(false); }
  1432→  };
  1433→
  1434→  return (
  1435→    <div className="flex flex-col gap-4">
  1436→      <div className="bg-card rounded-[14px] border border-border p-5 flex flex-col gap-3.5">
  1437→        <h3 className="text-[15px] font-bold">قيد يومية جديد</h3>
  1438→        <div className="grid grid-cols-[repeat(auto-fit,minmax(200px,1fr))] gap-3">
  1439→          <div><label className={labelStyle}>التاريخ</label><input type="date" value={date} onChange={(e) => setDate(e.target.value)} className={inputStyle} dir="ltr" /></div>
  1440→          <div><label className={labelStyle}>المرجع</label><input value={reference} onChange={(e) => setReference(e.target.value)} className={inputStyle} dir="ltr" /></div>
  1441→          <div><label className={labelStyle}>الحالة</label>
  1442→            <select value={status} onChange={(e) => setStatus(e.target.value)} className={inputStyle}><option value="draft">مسودة</option><option value="posted">مُرحّل</option></select>
  1443→          </div>
  1444→        </div>
  1445→        <div><label className={labelStyle}>الوصف</label><input value={description} onChange={(e) => setDescription(e.target.value)} className={inputStyle} /></div>
  1446→        <div>
  1447→          <div className="flex justify-between items-center mb-2">
  1448→            <label className={cn(labelStyle, "mb-0")}>البنود</label>
  1449→            <button onClick={addLine} className="bg-accent text-accent-foreground border border-border rounded-[6px] py-1 px-2.5 text-[11px] font-bold cursor-pointer inline-flex items-center gap-1"><Plus size={12} /> إضافة</button>
  1450→          </div>
  1451→          <div className="flex flex-col gap-2">
  1452→            {lines.map((l, i) => (
  1453→              <div key={i} className="grid grid-cols-[1fr_100px_100px_32px] gap-2 items-center">
  1454→                <select value={l.accountId ?? ""} onChange={(e) => updateLine(i, "accountId", Number(e.target.value))} className={inputStyle}><option value="">— اختر حساب —</option>{accounts.map((a) => <option key={a.id} value={a.id}>{a.code} — {a.nameAr}</option>)}</select>
  1455→                <input type="number" placeholder="مدين" value={l.debit} onChange={(e) => updateLine(i, "debit", Number(e.target.value))} className={inputStyle} dir="ltr" />
  1456→                <input type="number" placeholder="دائن" value={l.credit} onChange={(e) => updateLine(i, "credit", Number(e.target.value))} className={inputStyle} dir="ltr" />
