  1540→/// Learning Engine — records when a human employee overrides the matcher's
  1541→/// chosen candidate (e.g. matcher picked candidates[0] but employee confirmed
  1542→/// candidates[1]). These overrides feed back into the `historicalMatch` evidence
  1543→/// signal so the matcher LEARNS from human corrections over time.
  1544→///
  1545→/// Lifecycle:
  1546→///   1. matcher returns candidates[0] as winner (productId=A)
  1547→///   2. employee reviews, decides productId=B (candidates[1]) is correct
  1548→///   3. POST /api/products/match-override records:
  1549→///        inputText, inputNormalized, fromProductId=A, toProductId=B,
  1550→///        chosenAlias, auditId (link to original ProductMatchAudit decision)
  1551→///   4. Next time the SAME normalized input is matched, the matcher sees a past
  1552→///      override and boosts confidence for productId=B's aliases.
  1553→///
  1554→/// `inputNormalized` is the Arabic-normalized form (via normalizeArabic) so that
  1555→/// minor spelling variations still hit the same override record. The compound
  1556→/// index `(companySlug, inputNormalized)` is the primary lookup path.
  1557→model MatchOverride {
  1558→  id              Int      @id @default(autoincrement())
  1559→  companySlug     String
  1560→  inputText       String // raw invoice line description that was overridden
  1561→  inputNormalized String // normalizeArabic(inputText) — fast exact re-lookup
  1562→  fromProductId   Int? // matcher's original winner (null if matcher said "new")
  1563→  toProductId     Int // product the employee chose instead
  1564→  chosenAlias     String? // alias text the employee confirmed
  1565→  auditId         Int? // link to ProductMatchAudit row of the original decision
  1566→  reason          String? // optional employee note ("brand mismatch", "wrong size")
  1567→  overriddenBy    String // user email/id
  1568→  createdAt       DateTime @default(now())
  1569→
  1570→  tenant Company @relation(fields: [companySlug], references: [slug], onDelete: Restrict)
