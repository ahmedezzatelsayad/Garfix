  1409→/// Learning Engine — records when a human employee overrides the matcher's
  1410→/// chosen candidate (e.g. matcher picked candidates[0] but employee confirmed
  1411→/// candidates[1]). These overrides feed back into the `historicalMatch` evidence
  1412→/// signal so the matcher LEARNS from human corrections over time.
  1413→///
  1414→/// Lifecycle:
  1415→///   1. matcher returns candidates[0] as winner (productId=A)
  1416→///   2. employee reviews, decides productId=B (candidates[1]) is correct
  1417→///   3. POST /api/products/match-override records:
  1418→///        inputText, inputNormalized, fromProductId=A, toProductId=B,
  1419→///        chosenAlias, auditId (link to original ProductMatchAudit decision)

  1421→///      override and boosts confidence for productId=B's aliases.

  1423→/// `inputNormalized` is the Arabic-normalized form (via normalizeArabic) so that

  1700→/// Learning Engine candidate — tracks repeated AI decisions that MAY be
  1701→/// promotable to deterministic rules (Phase 11).
  1702→///
  1703→/// When the same normalized input produces the same AI output N times
  1704→/// (sampleCount >= threshold) with high consistency (confidence >= 0.95),
  1705→/// the candidate is promoted to status='promoted' and a Rule is created.
  1706→model RuleCandidate {
  1707→  id                Int      @id @default(autoincrement())
  1708→  companySlug       String
  1709→  requestType       String   // ocr | whatsapp | financial_analysis | matching | other
  1710→  patternSignature  String   // hash of normalized inputs grouped together
  1711→  sampleCount       Int      @default(0)
  1712→  consistentOutput  String   // JSON — the output that appeared in >= 95% of samples
  1713→  confidence        Float    @default(0)  // 0-1, percentage of samples matching consistentOutput
  1714→  status            String   @default("observing") // observing | promoted | rejected
  1715→  createdAt         DateTime @default(now())
  1716→  updatedAt         DateTime @updatedAt
  1717→
  1718→  @@index([companySlug, status])
  1719→  @@index([companySlug, patternSignature])
  1720→  @@index([status, confidence])
  1721→  @@map("rule_candidates")
  1722→}

  1724→/// Cross-company de-identified patterns (Phase 12).
  1725→///
  1726→/// Stores GENERAL patterns (e.g. "Apple Charger → SKU/VAT/Category") stripped
  1727→/// of all company-specific data. NO financial data, NO customer data, NO
  1728→/// company-identifying information. Only the product name/classification.
  1729→///
  1730→/// Privacy guarantee: no field in this table can be traced back to any

  1732→model GlobalPattern {

  1736→  suggestedVatCategory     String?  // suggested VAT category
  1737→  suggestedCategory        String?  // JSON — suggested product category
  1738→  contributingCompaniesCount Int     @default(0) // COUNT ONLY, no company IDs
  1739→  confidence               Float    @default(0)
  1740→  lastUpdated              DateTime @updatedAt
  1741→
  1742→  @@unique([patternKey])
  1743→  @@index([confidence])
  1744→  @@index([lastUpdated])

  1770→/// AI Compiler — compiled rules distilled from repeated AI patterns (Phase 16).
  1771→///
  1772→/// When the Learning Engine identifies a cluster of AI decisions that can be
  1773→/// replaced by deterministic logic (regex, decision tree, extractor), this
  1774→/// table stores the compiled rule PENDING HUMAN REVIEW. No compiled rule
  1775→/// is ever activated without explicit human approval (reviewedBy + reviewedAt).
  1776→///
  1777→/// This is the most critical table in the Economics Layer — a wrong compiled
  1778→/// rule could silently corrupt business decisions. Human-in-the-loop is
  1779→/// MANDATORY for status transitions from pending_review → active.
  1780→model CompiledRule {
  1781→  id                    Int      @id @default(autoincrement())
  1782→  requestType           String   // ocr | whatsapp | financial_analysis | matching | other
  1783→  patternDescription    String   // human-readable description of the pattern
  1784→  logicType             String   // regex | decision_tree | extractor
  1785→  logicDefinition       String   // JSON — the actual compiled logic (regex, tree, etc.)
  1786→  accuracyOnHistoricalSample Float @default(0) // measured accuracy (0-1)
  1787→  status                String   @default("pending_review") // pending_review | active | rejected
  1788→  reviewedBy            String?  // email of the human who approved/rejected
  1789→  reviewedAt            DateTime?
  1790→  sourceCandidateId     Int?     // link to RuleCandidate that generated this
  1791→  createdAt             DateTime @default(now())
  1792→  updatedAt             DateTime @updatedAt
  1793→
  1794→  @@index([status])
  1795→  @@index([requestType, status])
  1796→  @@index([accuracyOnHistoricalSample])
  1797→  @@map("compiled_rules")
  1798→}
