    18→// ─── Helper: count files matching pattern ────────────────────────────────────
    19→
    20→function countFiles(dir: string, pattern: RegExp, excludePattern?: RegExp): number {
    21→  let count = 0;
    22→  function walk(d: string) {
    23→    try {
    24→      for (const entry of readdirSync(d)) {
    25→        const full = join(d, entry);
    26→        const st = statSync(full);
    27→        if (st.isDirectory()) {
    28→          if (!excludePattern || !excludePattern.test(full)) walk(full);
    29→        } else if (pattern.test(entry)) {
    30→          if (!excludePattern || !excludePattern.test(full)) count++;
    31→        }
    32→      }
    33→    } catch { /* skip */ }
    34→  }
    35→  walk(dir);
    36→  return count;
    37→}
    38→
    39→function grepCount(file: string, pattern: RegExp): number {
    40→  const content = readFileSync(file, "utf-8");
    41→  const matches = content.match(pattern);
    42→  return matches ? matches.length : 0;
    43→}
    44→
    45→function grepLines(file: string, pattern: RegExp): string[] {
    46→  const content = readFileSync(file, "utf-8");
    47→  const lines = content.split("\n");
    48→  return lines.filter((l, i) => pattern.test(l)).map((l, i) => l);
    49→}
    50→
    51→function fileContains(file: string, pattern: RegExp): boolean {
    52→  try {
    53→    const content = readFileSync(file, "utf-8");
    54→    return pattern.test(content);
    55→  } catch { return false; }
    56→}
