   363→  });

   366→    const secret = "test-secret-key";
   367→    const payload = JSON.stringify({ event: "invoice.created" });
   368→    expect(verifyWebhookSignature(payload, "sha256=wrong_signature", secret)).toBe(false);
   369→  });
   370→
   371→  it("rejects signature with wrong secret", () => {
   372→    const secret = "correct-secret";
   373→    const wrongSecret = "wrong-secret";
   374→    const payload = JSON.stringify({ event: "test" });
   375→    const sig = crypto.createHmac("sha256", secret).update(payload).digest("hex");
   376→    expect(verifyWebhookSignature(payload, `sha256=${sig}`, wrongSecret)).toBe(false);
   377→  });
   378→
   379→  it("rejects signature without sha256= prefix", () => {
   380→    const secret = "test-secret";
   381→    const payload = JSON.stringify({ event: "test" });
   382→    const sig = crypto.createHmac("sha256", secret).update(payload).digest("hex");
   383→    expect(verifyWebhookSignature(payload, sig, secret)).toBe(false);
   384→  });
