    68→  const [form, setForm] = useState({
    69→    name: "", nameAr: "", emoji: "", color: "#7c3aed",
    70→    phone: "", email: "", address: "", vatNumber: "",
    71→    commercialRegistration: "",
    72→    currency: "KWD", country: "KW", defaultTaxRate: "0",

    82→

    97→        vatNumber: activeCompany.vatNumber || "",

   100→        country: activeCompany.country || "KW",

   121→            primaryColor: s.primaryColor || "#7C3AED",

   123→            fontSize: s.fontSize || 12,
   124→            showLogo: s.showLogo ?? true,
   125→            logoPosition: s.logoPosition || "right",
   126→            showPaymentInfo: s.showPaymentInfo ?? true,
   127→            showStamp: s.showStamp ?? false,
   128→            invoiceTypes: s.invoiceTypes ? s.invoiceTypes.split(",").filter(Boolean) : ["sales", "purchase", "quote"],

   131→          setTemplateForm(defaultTemplateSettings);
   132→        }
   133→        setTemplateLoaded(true);

   137→          setTemplateForm(defaultTemplateSettings);

   152→        method: "PATCH",
   153→        headers: { "Content-Type": "application/json" },
   154→        body: JSON.stringify(form),
   155→      });

   157→        const e = await res.json().catch(() => ({}));
   158→        throw new Error(e.error || "Failed");

   160→      toast.success("تم حفظ الإعدادات");
   161→      onUpdated();

   163→      toast.error(err instanceof Error ? err.message : "خطأ");

   165→      setSaving(false);
   166→    }

   177→          companySlug: activeCompany.slug,

   179→        }),

   182→        const e = await res.json().catch(() => ({}));

   187→      toast.error(err instanceof Error ? err.message : "خطأ");

   321→          </div>

   464→                    onChange={(e) => setTemplateForm((p) => ({ ...p, fontSize: Number(e.target.value) }))}
   465→                    className="w-full accent-primary"
   466→                  />
   467→                </div>
   468→              </div>
   469→
   470→              {/* ── Toggle switches ───────────────────────────────────── */}
   471→              <div className="grid grid-cols-1 sm:grid-cols-3 gap-4">
   472→                {/* Show Logo */}
   473→                <div className="flex items-center justify-between rounded-lg border border-border p-3">
   474→                  <Label className="flex items-center gap-1.5 text-xs font-semibold text-muted-foreground cursor-pointer">
   475→                    <ImageIcon size={14} /> إظهار الشعار
   476→                  </Label>
   477→                  <Switch
   478→                    checked={templateForm.showLogo}
   479→                    onCheckedChange={(v) => setTemplateForm((p) => ({ ...p, showLogo: v }))}
   480→                  />
   481→                </div>
   482→
   483→                {/* Show Payment Info */}
   484→                <div className="flex items-center justify-between rounded-lg border border-border p-3">
   485→                  <Label className="flex items-center gap-1.5 text-xs font-semibold text-muted-foreground cursor-pointer">
   486→                    <CreditCard size={14} /> معلومات الدفع
   487→                  </Label>
   488→                  <Switch
   489→                    checked={templateForm.showPaymentInfo}
   490→                    onCheckedChange={(v) => setTemplateForm((p) => ({ ...p, showPaymentInfo: v }))}
   491→                  />
   492→                </div>
   493→
   494→                {/* Show Stamp */}
   495→                <div className="flex items-center justify-between rounded-lg border border-border p-3">
   496→                  <Label className="flex items-center gap-1.5 text-xs font-semibold text-muted-foreground cursor-pointer">
   497→                    <Stamp size={14} /> إظهار الختم
   498→                  </Label>
   499→                  <Switch
   500→                    checked={templateForm.showStamp}
   501→                    onCheckedChange={(v) => setTemplateForm((p) => ({ ...p, showStamp: v }))}
   502→                  />
   503→                </div>
   504→              </div>
   505→
   506→              {/* ── Logo position ─────────────────────────────────────── */}
   507→              <div className="space-y-1.5">
   508→                <Label className="text-xs font-semibold text-muted-foreground">موضع الشعار</Label>
   509→                <div className="flex gap-2">
   510→                  {(["right", "center", "left"] as const).map((pos) => {
   511→                    const labels: Record<string, string> = { right: "يمين", center: "وسط", left: "يسار" };
   512→                    return (
   513→                      <button
   514→                        key={pos}
   515→                        type="button"
   516→                        onClick={() => setTemplateForm((p) => ({ ...p, logoPosition: pos }))}
   517→                        className={`flex-1 rounded-lg border-2 py-2 text-sm font-medium transition-all cursor-pointer ${
   518→                          templateForm.logoPosition === pos
   519→                            ? "border-primary bg-primary/5 text-primary"
   520→                            : "border-border hover:border-primary/40"
   521→                        }`}
   522→                      >
   523→                        {labels[pos]}
   524→                      </button>
   525→                    );
   526→                  })}
   527→                </div>
   528→              </div>
   529→
   530→              {/* ── Invoice types multi-select ─────────────────────────── */}
   531→              <div className="space-y-2">
   532→                <Label className="text-xs font-semibold text-muted-foreground">أنواع الفواتير المطبّق عليها القالب</Label>
   533→                <div className="flex flex-wrap gap-3">
   534→                  {INVOICE_TYPE_OPTIONS.map((opt) => (
   535→                    <label
   536→                      key={opt.id}
   537→                      className="flex items-center gap-2 cursor-pointer rounded-lg border border-border px-3 py-2 transition-all hover:bg-muted/50 has-[button[data-state=checked]]:border-primary has-[button[data-state=checked]]:bg-primary/5"
   538→                    >
   539→                      <Checkbox
   540→                        checked={templateForm.invoiceTypes.includes(opt.id)}
   541→                        onCheckedChange={() => toggleInvoiceType(opt.id)}
   542→                      />
   543→                      <span className="text-sm">{opt.label}</span>
   544→                    </label>
   545→                  ))}
   546→                </div>
   547→                {templateForm.invoiceTypes.length === 0 && (
   548→                  <p className="text-xs text-destructive">يجب اختيار نوع فاتورة واحد على الأقل</p>
   549→                )}
   550→              </div>
   551→
   552→              {/* ── Save button ───────────────────────────────────────── */}
   553→              <div className="flex justify-end pt-2">
   554→                <Button
   555→                  onClick={saveTemplateSettings}
   556→                  disabled={savingTemplate || templateForm.invoiceTypes.length === 0}
   557→                  className="gap-2"
   558→                >
   559→                  <Save size={14} />
   560→                  {savingTemplate ? "جارٍ الحفظ…" : "حفظ إعدادات القالب"}
   561→                </Button>
   562→              </div>
   563→            </>
   564→          )}
   565→        </CardContent>
   566→      </Card>
   567→    </div>
   568→  );
   569→}
