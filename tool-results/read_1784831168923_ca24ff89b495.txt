     1→// @ts-nocheck
     2→/**
     3→ * invoices-crud.test.ts — Tests for the invoice REST route handlers.
     4→ *
     5→ * Covers:
     6→ *   POST   /api/invoices            (create)
     7→ *   GET    /api/invoices            (list)
     8→ *   GET    /api/invoices/[id]       (single)
     9→ *   PATCH  /api/invoices/[id]       (update with optimistic lock)
    10→ *   DELETE /api/invoices/[id]       (soft delete)
    11→ *
    12→ * Strategy:
    13→ *  - Mock @/lib/auth, @/lib/middleware, @/lib/audit, @/lib/usageMeter via
    14→ *    mock.module — these modules are only imported by the invoice route
    15→ *    handlers (no other test file imports them), so the mocks don't leak.
    16→ *  - Do NOT mock @/lib/inventorySync or @/lib/db via mock.module. Bun's
    17→ *    mock.module is global across test files by default; mocking those
    18→ *    modules would leak into collision-recovery-audit.test.ts (which needs
    19→ *    the real syncInventoryOnSale) and productMatcher.test.ts (which needs
    20→ *    the real matchProduct via its own @/lib/db mock). Instead, we
    21→ *    monkey-patch the shared `db` object's properties (invoice, auditLog,
    22→ *    company, $transaction, featureFlag, platformSetting) in beforeAll and
    23→ *    restore them in afterAll.
    24→ *  - The real syncInventoryOnSale is exercised on the POST happy path via
    25→ *    db.$transaction, which passes a rich fake `tx` that supports the sync
    26→ *    function's full call surface (warehouse, productAlias, productCatalog,
    27→ *    inventoryItem, stockMovement, productMatchAudit).
    28→ *  - Each test sets the mutable state (authUser, existingInvoice, etc.) it
    29→ *    needs, then constructs a fake `NextRequest` via `new NextRequest(url,
    30→ *    { method, body, headers })` and invokes the exported handler directly.
    31→ *  - For dynamic [id] routes we pass `{ params: Promise.resolve({ id }) }`
    32→ *    because the route signature is
    33→ *    `(req: NextRequest, { params }: { params: Promise<{ id: string }> })`.
    34→ *
    35→ * The mocks intentionally replicate the behavior of the real auth/tenant
    36→ * helpers (assertCompanyAccess uses user.companies.includes(slug), admins
    37→ * bypass, etc.) so the security-critical assertions (cross-tenant 403)
    38→ * exercise the same logic the production code relies on.
    39→ */

    70→const invoiceFindUnique = mock(async (args: any) => {

    81→  return { id: args.where?.id ?? 0, ...args.data };
    82→});
    83→const invoiceUpdateMany = mock(async (args: any) => {
    84→  lastUpdateManyArgs = args;
    85→  // If the test forced count=0 (simulating a concurrent edit conflict), honor it.
    86→  if (nextUpdateManyCount !== null) {
    87→    const c = nextUpdateManyCount;
    88→    nextUpdateManyCount = null;
    89→    return { count: c };
    90→  }
    91→  // Simulate the atomic version check: if the where clause includes a version
    92→  // filter that doesn't match the current invoiceById.version, return count=0.
    93→  if (
    94→    args.where &&
    95→    args.where.version !== undefined &&
    96→    invoiceById &&
    97→    args.where.version !== invoiceById.version
    98→  ) {
    99→    return { count: 0 };
   100→  }
   101→  // Simulate the atomic soft-delete check: if the row is soft-deleted, return 0.
   102→  if (invoiceById && invoiceById.deletedAt && args.where && args.where.deletedAt === null) {
   103→    return { count: 0 };
   104→  }
   105→  return { count: 1 };
   106→});

   120→  if (!w) return null;

   124→    companySlug: w.companySlug,

   126→    key: w.key,
