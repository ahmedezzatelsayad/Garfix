    14→// SECTION 0: Seeded PRNG (mulberry32) — deterministic, reproducible
    15→// ═══════════════════════════════════════════════════════════════════════════════
    16→
    17→export class SeededRandom {
    18→  private state: number;
    19→
    20→  constructor(seed: number = 42) {
    21→    this.state = seed | 0;
    22→  }
    23→
    24→  /** Returns float in [0, 1) */
    25→  next(): number {
    26→    let t = (this.state += 0x6d2b79f5);
    27→    t = Math.imul(t ^ (t >>> 15), t | 1);
    28→    t ^= t + Math.imul(t ^ (t >>> 7), t | 61);
    29→    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
    30→  }
    31→
    32→  /** Returns integer in [min, max] inclusive */
    33→  int(min: number, max: number): number {
    34→    return Math.floor(this.next() * (max - min + 1)) + min;
    35→  }
    36→
    37→  /** Returns float in [min, max) */
    38→  float(min: number, max: number): number {
    39→    return this.next() * (max - min) + min;
    40→  }
    41→
    42→  /** Pick a random element from an array */
    43→  pick<T>(arr: T[]): T {
    44→    return arr[Math.floor(this.next() * arr.length)];
    45→  }
    46→
    47→  /** Pick N unique elements from array */
    48→  pickN<T>(arr: T[], n: number): T[] {
    49→    const copy = [...arr];
    50→    const result: T[] = [];
    51→    for (let i = 0; i < Math.min(n, copy.length); i++) {
    52→      const idx = Math.floor(this.next() * copy.length);
    53→      result.push(copy.splice(idx, 1)[0]);
    54→    }
    55→    return result;
    56→  }
    57→
    58→  /** Shuffle array in place */
    59→  shuffle<T>(arr: T[]): T[] {
    60→    for (let i = arr.length - 1; i > 0; i--) {
    61→      const j = Math.floor(this.next() * (i + 1));
    62→      [arr[i], arr[j]] = [arr[j], arr[i]];
    63→    }
    64→    return arr;
    65→  }
    66→
    67→  /** Boolean with probability p (default 0.5) */
    68→  bool(p: number = 0.5): boolean {
    69→    return this.next() < p;
    70→  }
    71→
    72→  /** Weighted random selection from an array of [item, weight] pairs */
    73→  weighted<T>(items: [T, number][]): T {
    74→    const total = items.reduce((s, [, w]) => s + w, 0);
    75→    let r = this.next() * total;
    76→    for (const [item, w] of items) {
    77→      r -= w;
    78→      if (r <= 0) return item;
    79→    }
    80→    return items[items.length - 1][0];
    81→  }
    82→
    83→  /** Generate a random date between start and end */
    84→  dateBetween(start: Date, end: Date): Date {
    85→    const ms = start.getTime() + this.next() * (end.getTime() - start.getTime());
    86→    return new Date(ms);
    87→  }
    88→}

   690→function cuid(): string {
   691→  const timestamp = Date.now().toString(36);
   692→  const random = Math.random().toString(36).substring(2, 10);
   693→  return `${timestamp}${random}`;
   694→}

   696→function slugify(text: string): string {
   697→  return text
   698→    .toLowerCase()
   699→    .replace(/\s+/g, '-')
   700→    .replace(/[^\w-]/g, '')
   701→    .substring(0, 50);
   702→}

   704→function formatMoney(amount: number, currency: Currency): string {
   705→  const decimals = currency === 'KWD' || currency === 'BHD' ? 3 : 2;
   706→  return amount.toFixed(decimals);
   707→}

   709→function randomDateRange(rng: SeededRandom, start: Date, end: Date): { issueDate: string; dueDate: string } {
   710→  const issue = rng.dateBetween(start, end);
   711→  const dueDays = rng.int(15, 90);
   712→  const due = new Date(issue);
   713→  due.setDate(due.getDate() + dueDays);
   714→  return {
   715→    issueDate: issue.toISOString().split('T')[0],
   716→    dueDate: due.toISOString().split('T')[0],
   717→  };
   718→}

  1401→
  1402→      const response = await fetch('https://openrouter.ai/api/v1/chat/completions', {
  1403→        method: 'POST',
  1404→        signal: controller.signal,

  1411→        body: JSON.stringify({

  1414→          max_tokens: 5,
  1415→          temperature: 0,

  1418→

  1421→      if (response.ok) {

  1425→    } catch {

  1572→export function calculateMetrics(companiesOrTelemetry: SyntheticCompany[] | TelemetryEntry[], telemetryOrCompanies?: SyntheticCompany[] | TelemetryEntry[]): MetricsSummary {
  1573→  // Allow calling as calculateMetrics(companies, telemetry) or calculateMetrics(telemetry, companies)
  1574→  // The first array argument that contains objects with a 'slug' property is the companies array
  1575→  let companies: SyntheticCompany[];
  1576→  let telemetry: TelemetryEntry[];
  1577→
  1578→  if (companiesOrTelemetry.length === 0) {
  1579→    companies = [];
  1580→    telemetry = [];
  1581→  } else {
  1582→    const first = companiesOrTelemetry[0] as any;
  1583→    if (first && ('slug' in first || 'invoices' in first)) {
  1584→      // First arg is companies
  1585→      companies = companiesOrTelemetry as SyntheticCompany[];
  1586→      telemetry = (telemetryOrCompanies as TelemetryEntry[]) ?? [];
  1587→    } else {
  1588→      // First arg is telemetry (swapped)
  1589→      telemetry = companiesOrTelemetry as TelemetryEntry[];
  1590→      companies = (telemetryOrCompanies as SyntheticCompany[]) ?? [];
  1591→    }
  1592→  }
  1593→  if (telemetry.length === 0) {
  1594→    return {
  1595→      totalRequests: 0,
  1596→      totalCompanies: 0,
  1597→      totalCostUsd: 0,
  1598→      providerDistribution: {},
  1599→      modelDistribution: {},
  1600→      totalTokenUsage: 0,
  1601→      totalUsdSpent: 0,
  1602→      avgCostPerRequest: 0,
  1603→      avgCostPerInvoice: 0,
  1604→      avgCostPerCompany: 0,
  1605→      highestCostTenants: [],
  1606→      cacheHitRate: 0,
  1607→      memoryHitRate: 0,
  1608→      ruleHitRate: 0,
  1609→      patternHitRate: 0,
  1610→      learningImprovement: { firstHalfAvgCost: 0, secondHalfAvgCost: 0, improvementPct: 0 },
  1611→      requestsPerMinute: 0,
  1612→      p50Latency: 0,
  1613→      p95Latency: 0,
  1614→      p99Latency: 0,
  1615→      errorRate: 0,
  1616→      budgetBlockedCount: 0,
  1617→    };
  1618→  }
  1619→
  1620→  const totalRequests = telemetry.length;
  1621→  const totalTokens = telemetry.reduce((s, e) => s + e.totalTokens, 0);
  1622→  const totalUsd = telemetry.reduce((s, e) => s + e.costUsd, 0);
  1623→  const totalInvoices = companies.reduce((s, c) => s + (c.invoices?.length ?? 0), 0) || 1;
  1624→
  1625→  // Provider distribution (with cost per provider)
  1626→  const providerDist: Record<string, { requests: number; cost: number }> = {};
  1627→  const modelDist: Record<string, { requests: number; cost: number }> = {};
  1628→  const tenantCosts: Map<string, { cost: number; requests: number }> = new Map();
  1629→
  1630→  for (const e of telemetry) {
  1631→    const pd = providerDist[e.provider] ?? { requests: 0, cost: 0 };
  1632→    pd.requests += 1;
  1633→    pd.cost += e.costUsd;
  1634→    providerDist[e.provider] = pd;
  1635→
  1636→    const md = modelDist[e.model] ?? { requests: 0, cost: 0 };
  1637→    md.requests += 1;
  1638→    md.cost += e.costUsd;
  1639→    modelDist[e.model] = md;
  1640→    const tc = tenantCosts.get(e.tenant) ?? { cost: 0, requests: 0 };
  1641→    tc.cost += e.costUsd;
  1642→    tc.requests += 1;
  1643→    tenantCosts.set(e.tenant, tc);
  1644→  }
  1645→
  1646→  // Hit rates (out of all requests, not just AI ones)
  1647→  const cacheHitRate = telemetry.filter(e => e.cacheHit).length / totalRequests;
  1648→  const memoryHitRate = telemetry.filter(e => e.memoryHit).length / totalRequests;
  1649→  const ruleHitRate = telemetry.filter(e => e.ruleHit).length / totalRequests;
  1650→  const patternHitRate = telemetry.filter(e => e.patternHit).length / totalRequests;
  1651→
  1652→  // Latency percentiles
  1653→  const sortedLatencies = telemetry.map(e => e.latencyMs).sort((a, b) => a - b);
  1654→  const p50 = sortedLatencies[Math.floor(sortedLatencies.length * 0.5)] ?? 0;
  1655→  const p95 = sortedLatencies[Math.floor(sortedLatencies.length * 0.95)] ?? 0;
  1656→  const p99 = sortedLatencies[Math.floor(sortedLatencies.length * 0.99)] ?? 0;
  1657→
  1658→  // Learning improvement: first half vs second half avg cost
  1659→  const mid = Math.floor(totalRequests / 2);
  1660→  const firstHalf = telemetry.slice(0, mid);
  1661→  const secondHalf = telemetry.slice(mid);
  1662→  const fhAvg = firstHalf.length > 0 ? firstHalf.reduce((s, e) => s + e.costUsd, 0) / firstHalf.length : 0;
  1663→  const shAvg = secondHalf.length > 0 ? secondHalf.reduce((s, e) => s + e.costUsd, 0) / secondHalf.length : 0;
  1664→  const improvementPct = fhAvg > 0 ? ((fhAvg - shAvg) / fhAvg) * 100 : 0;
  1665→
  1666→  // Error rate
  1667→  const errorRate = telemetry.filter(e => (e.errors?.length ?? 0) > 0).length / totalRequests;
  1668→
  1669→  // Budget blocked
  1670→  const budgetBlocked = telemetry.filter(e => e.recoveryPath === 'manual_review' && (e.errors ?? []).includes('budget_exceeded')).length;
  1671→
  1672→  // Highest cost tenants (top 20)
  1673→  const highestCostTenants = Array.from(tenantCosts.entries())
  1674→    .sort((a, b) => b[1].cost - a[1].cost)
  1675→    .slice(0, 20)
  1676→    .map(([tenant, data]) => ({ tenant, cost: data.cost, requests: data.requests }));
  1677→
  1678→  // Requests per minute (assuming 1 month = 43200 minutes for generated data)
  1679→  const rqsPerMin = totalRequests / 43200;
  1680→
  1681→  return {
  1682→    totalRequests,
  1683→    totalCompanies: companies.length,
  1684→    totalCostUsd: totalUsd,
  1685→    providerDistribution: providerDist,
  1686→    modelDistribution: modelDist,
  1687→    totalTokenUsage: totalTokens,
  1688→    totalUsdSpent: totalUsd,
  1689→    avgCostPerRequest: totalUsd / totalRequests,
  1690→    avgCostPerInvoice: totalUsd / totalInvoices,
  1691→    avgCostPerCompany: totalUsd / companies.length,
  1692→    highestCostTenants,
  1693→    cacheHitRate,
  1694→    memoryHitRate,
  1695→    ruleHitRate,
  1696→    patternHitRate,
  1697→    learningImprovement: {
  1698→      firstHalfAvgCost: fhAvg,
  1699→      secondHalfAvgCost: shAvg,
  1700→      improvementPct,
  1701→    },
  1702→    requestsPerMinute: rqsPerMin,
  1703→    p50Latency: p50,
  1704→    p95Latency: p95,
  1705→    p99Latency: p99,
  1706→    errorRate,
  1707→    budgetBlockedCount: budgetBlocked,
  1708→  };
  1709→}
