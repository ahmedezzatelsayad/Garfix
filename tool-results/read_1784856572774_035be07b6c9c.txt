    14→// SECTION 0: Seeded PRNG (mulberry32) — deterministic, reproducible
    15→// ═══════════════════════════════════════════════════════════════════════════════
    16→
    17→export class SeededRandom {
    18→  private state: number;
    19→
    20→  constructor(seed: number = 42) {
    21→    this.state = seed | 0;
    22→  }
    23→
    24→  /** Returns float in [0, 1) */
    25→  next(): number {
    26→    let t = (this.state += 0x6d2b79f5);
    27→    t = Math.imul(t ^ (t >>> 15), t | 1);
    28→    t ^= t + Math.imul(t ^ (t >>> 7), t | 61);
    29→    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
    30→  }
    31→
    32→  /** Returns integer in [min, max] inclusive */
    33→  int(min: number, max: number): number {
    34→    return Math.floor(this.next() * (max - min + 1)) + min;
    35→  }
    36→
    37→  /** Returns float in [min, max) */
    38→  float(min: number, max: number): number {
    39→    return this.next() * (max - min) + min;
    40→  }
    41→
    42→  /** Pick a random element from an array */
    43→  pick<T>(arr: T[]): T {
    44→    return arr[Math.floor(this.next() * arr.length)];
    45→  }
    46→
    47→  /** Pick N unique elements from array */
    48→  pickN<T>(arr: T[], n: number): T[] {
    49→    const copy = [...arr];
    50→    const result: T[] = [];
    51→    for (let i = 0; i < Math.min(n, copy.length); i++) {
    52→      const idx = Math.floor(this.next() * copy.length);
    53→      result.push(copy.splice(idx, 1)[0]);
    54→    }
    55→    return result;
    56→  }
    57→
    58→  /** Shuffle array in place */
    59→  shuffle<T>(arr: T[]): T[] {
    60→    for (let i = arr.length - 1; i > 0; i--) {
    61→      const j = Math.floor(this.next() * (i + 1));
    62→      [arr[i], arr[j]] = [arr[j], arr[i]];
    63→    }
    64→    return arr;
    65→  }
    66→
    67→  /** Boolean with probability p (default 0.5) */
    68→  bool(p: number = 0.5): boolean {
    69→    return this.next() < p;
    70→  }
    71→
    72→  /** Weighted random selection from an array of [item, weight] pairs */
    73→  weighted<T>(items: [T, number][]): T {
    74→    const total = items.reduce((s, [, w]) => s + w, 0);
    75→    let r = this.next() * total;
    76→    for (const [item, w] of items) {
    77→      r -= w;
    78→      if (r <= 0) return item;
    79→    }
    80→    return items[items.length - 1][0];
    81→  }
    82→
    83→  /** Generate a random date between start and end */
    84→  dateBetween(start: Date, end: Date): Date {
    85→    const ms = start.getTime() + this.next() * (end.getTime() - start.getTime());
    86→    return new Date(ms);
    87→  }
    88→}

   690→function cuid(): string {
   691→  const timestamp = Date.now().toString(36);
   692→  const random = Math.random().toString(36).substring(2, 10);
   693→  return `${timestamp}${random}`;
   694→}

   696→function slugify(text: string): string {
   697→  return text
   698→    .toLowerCase()
   699→    .replace(/\s+/g, '-')
   700→    .replace(/[^\w-]/g, '')
   701→    .substring(0, 50);
   702→}

   704→function formatMoney(amount: number, currency: Currency): number {
   705→  const decimals = currency === 'KWD' || currency === 'BHD' ? 3 : 2;
   706→  return parseFloat(amount.toFixed(decimals));
   707→}

   709→function randomDateRange(rng: SeededRandom, start: Date, end: Date): { issueDate: string; dueDate: string } {
   710→  const issue = rng.dateBetween(start, end);
   711→  const dueDays = rng.int(15, 90);
   712→  const due = new Date(issue);
   713→  due.setDate(due.getDate() + dueDays);
   714→  return {
   715→    issueDate: issue.toISOString().split('T')[0],
   716→    dueDate: due.toISOString().split('T')[0],
   717→  };
   718→}

  1344→export async function callOpenRouter(
  1345→  apiKey: string,
  1346→  prompt: string,
  1347→  model?: string,
  1348→  useFastest?: boolean,
  1349→): Promise<OpenRouterResponse> {
  1350→  const selectedModel = useFastest
  1351→    ? await selectFastestModel(apiKey)
  1352→    : model ?? 'deepseek/deepseek-chat';
  1353→
  1354→  const startMs = Date.now();
  1355→
  1356→  const response = await fetch('https://openrouter.ai/api/v1/chat/completions', {
  1357→    method: 'POST',
  1358→    headers: {
  1359→      'Authorization': `Bearer ${apiKey}`,
  1360→      'Content-Type': 'application/json',
  1361→      'HTTP-Referer': 'https://garfix.app/founder-validation',
  1362→      'X-Title': 'GarfiX Founder Validation Suite',

  1364→    body: JSON.stringify({
  1365→      model: selectedModel,
  1366→      messages: [
  1367→        {
  1368→          role: 'system',
  1369→          content: 'You are a production validation assistant for GarfiX invoice management SaaS. Respond concisely.',
  1370→        },
  1371→        { role: 'user', content: prompt },
  1372→      ],
  1373→      max_tokens: 500,
  1374→      temperature: 0.3,
  1375→    }),
  1376→  });
  1377→
  1378→  if (!response.ok) {
  1379→    const errorBody = await response.text();
  1380→    throw new Error(`OpenRouter API error ${response.status}: ${errorBody}`);
  1381→  }
  1382→
  1383→  const data = await response.json() as OpenRouterResponse;
  1384→  const latencyMs = Date.now() - startMs;
  1385→
  1386→  // Enrich with latency metadata (returned via data but we also track it)
  1387→  void latencyMs;
  1388→
  1389→  return data;
  1390→}

  1392→/**
  1393→ * selectFastestModel — Pings each model in speed order with a trivial prompt
  1394→ * and returns the ID of the first one that responds within 5 seconds.
  1395→ */
  1396→export async function selectFastestModel(apiKey: string): Promise<string> {
  1397→  const testPrompt = 'Reply with exactly: OK';
  1398→
  1399→  for (const modelId of MODELS_BY_SPEED) {
  1400→    try {
  1401→      const startMs = Date.now();
  1402→      const controller = new AbortController();
  1403→      const timeout = setTimeout(() => controller.abort(), 5000);
  1404→
  1405→      const response = await fetch('https://openrouter.ai/api/v1/chat/completions', {
  1406→        method: 'POST',
  1407→        signal: controller.signal,
  1408→        headers: {
  1409→          'Authorization': `Bearer ${apiKey}`,
  1410→          'Content-Type': 'application/json',
  1411→          'HTTP-Referer': 'https://garfix.app/founder-validation',
  1412→          'X-Title': 'GarfiX Model Selection',
  1413→        },
  1414→        body: JSON.stringify({
  1415→          model: modelId,
  1416→          messages: [{ role: 'user', content: testPrompt }],
  1417→          max_tokens: 5,
  1418→          temperature: 0,
  1419→        }),
  1420→      });
  1421→
  1422→      clearTimeout(timeout);
  1423→
  1424→      if (response.ok) {
  1425→        const latencyMs = Date.now() - startMs;
  1426→        return modelId; // First successful = fastest (ordered by expected speed)
  1427→      }
  1428→    } catch {
  1429→      // Model unavailable or timeout, try next
  1430→    }
  1431→  }
  1432→
  1433→  // Fallback

  1435→}

  1575→/**

  1587→ * Migration plan (v13):

  1592→export function calculateMetrics(companiesOrTelemetry: SyntheticCompany[] | TelemetryEntry[], telemetryOrCompanies?: SyntheticCompany[] | TelemetryEntry[]): MetricsSummary {
  1593→  // Allow calling as calculateMetrics(companies, telemetry) or calculateMetrics(telemetry, companies)
  1594→  // The first array argument that contains objects with a 'slug' property is the companies array
  1595→  let companies: SyntheticCompany[];
  1596→  let telemetry: TelemetryEntry[];
  1597→
  1598→  if (companiesOrTelemetry.length === 0) {
  1599→    companies = [];
  1600→    telemetry = [];
  1601→  } else {
  1602→    const first = companiesOrTelemetry[0] as any;
  1603→    if (first && ('slug' in first || 'invoices' in first)) {
  1604→      // First arg is companies
  1605→      companies = companiesOrTelemetry as SyntheticCompany[];
  1606→      telemetry = (telemetryOrCompanies as TelemetryEntry[]) ?? [];
  1607→    } else {
  1608→      // First arg is telemetry (swapped)
  1609→      telemetry = companiesOrTelemetry as TelemetryEntry[];
  1610→      companies = (telemetryOrCompanies as SyntheticCompany[]) ?? [];
  1611→    }
  1612→  }
  1613→  if (telemetry.length === 0) {
  1614→    return {
  1615→      totalRequests: 0,
  1616→      totalCompanies: 0,
  1617→      totalCostUsd: 0,
  1618→      providerDistribution: {},
  1619→      modelDistribution: {},
  1620→      totalTokenUsage: 0,
  1621→      totalUsdSpent: 0,
  1622→      avgCostPerRequest: 0,
  1623→      avgCostPerInvoice: 0,
  1624→      avgCostPerCompany: 0,
  1625→      highestCostTenants: [],
  1626→      cacheHitRate: 0,
  1627→      memoryHitRate: 0,
  1628→      ruleHitRate: 0,
  1629→      patternHitRate: 0,
  1630→      learningImprovement: { firstHalfAvgCost: 0, secondHalfAvgCost: 0, improvementPct: 0 },
  1631→      requestsPerMinute: 0,
  1632→      p50Latency: 0,
  1633→      p95Latency: 0,
  1634→      p99Latency: 0,
  1635→      errorRate: 0,
  1636→      budgetBlockedCount: 0,
  1637→    };
  1638→  }
  1639→
  1640→  const totalRequests = telemetry.length;
  1641→  const totalTokens = telemetry.reduce((s, e) => s + e.totalTokens, 0);
  1642→  const totalUsd = telemetry.reduce((s, e) => s + e.costUsd, 0);
  1643→  // P0 FIX: Remove `|| 1` fallback — when totalInvoices=0, avgCostPerInvoice
  1644→  // should return 0, not totalUsd/1. The `|| 1` caused calculateMetrics(entries, [])
  1645→  // to return avgCostPerInvoice=1 instead of 0.
  1646→  const totalInvoices = companies.reduce((s, c) => s + (c.invoices?.length ?? 0), 0);
  1647→
  1648→  // Provider distribution (with cost per provider)
  1649→  const providerDist: Record<string, { requests: number; cost: number }> = {};
  1650→  const modelDist: Record<string, { requests: number; cost: number }> = {};
  1651→  const tenantCosts: Map<string, { cost: number; requests: number }> = new Map();
  1652→
  1653→  for (const e of telemetry) {
  1654→    const pd = providerDist[e.provider] ?? { requests: 0, cost: 0 };
  1655→    pd.requests += 1;
  1656→    pd.cost += e.costUsd;
  1657→    providerDist[e.provider] = pd;
  1658→
  1659→    const md = modelDist[e.model] ?? { requests: 0, cost: 0 };
  1660→    md.requests += 1;
  1661→    md.cost += e.costUsd;
  1662→    modelDist[e.model] = md;
  1663→    const tc = tenantCosts.get(e.tenant) ?? { cost: 0, requests: 0 };
  1664→    tc.cost += e.costUsd;
  1665→    tc.requests += 1;
  1666→    tenantCosts.set(e.tenant, tc);
  1667→  }
  1668→
  1669→  // Hit rates (out of all requests, not just AI ones)
  1670→  const cacheHitRate = telemetry.filter(e => e.cacheHit).length / totalRequests;
  1671→  const memoryHitRate = telemetry.filter(e => e.memoryHit).length / totalRequests;
  1672→  const ruleHitRate = telemetry.filter(e => e.ruleHit).length / totalRequests;
  1673→  const patternHitRate = telemetry.filter(e => e.patternHit).length / totalRequests;
  1674→
  1675→  // Latency percentiles
  1676→  const sortedLatencies = telemetry.map(e => e.latencyMs).sort((a, b) => a - b);
  1677→  const p50 = sortedLatencies[Math.floor(sortedLatencies.length * 0.5)] ?? 0;
  1678→  const p95 = sortedLatencies[Math.floor(sortedLatencies.length * 0.95)] ?? 0;
  1679→  const p99 = sortedLatencies[Math.floor(sortedLatencies.length * 0.99)] ?? 0;
  1680→
  1681→  // Learning improvement: first half vs second half avg cost
  1682→  const mid = Math.floor(totalRequests / 2);
  1683→  const firstHalf = telemetry.slice(0, mid);
  1684→  const secondHalf = telemetry.slice(mid);
  1685→  const fhAvg = firstHalf.length > 0 ? firstHalf.reduce((s, e) => s + e.costUsd, 0) / firstHalf.length : 0;
  1686→  const shAvg = secondHalf.length > 0 ? secondHalf.reduce((s, e) => s + e.costUsd, 0) / secondHalf.length : 0;
  1687→  const improvementPct = fhAvg > 0 ? ((fhAvg - shAvg) / fhAvg) * 100 : 0;
  1688→
  1689→  // Error rate
  1690→  const errorRate = telemetry.filter(e => (e.errors?.length ?? 0) > 0).length / totalRequests;
  1691→
  1692→  // Budget blocked
  1693→  const budgetBlocked = telemetry.filter(e => e.recoveryPath === 'manual_review' && (e.errors ?? []).includes('budget_exceeded')).length;
  1694→
  1695→  // Highest cost tenants (top 20)
  1696→  const highestCostTenants = Array.from(tenantCosts.entries())
  1697→    .sort((a, b) => b[1].cost - a[1].cost)
  1698→    .slice(0, 20)
  1699→    .map(([tenant, data]) => ({ tenant, cost: data.cost, requests: data.requests }));
  1700→
  1701→  // Requests per minute (assuming 1 month = 43200 minutes for generated data)
  1702→  const rqsPerMin = totalRequests / 43200;
  1703→
  1704→  return {
  1705→    totalRequests,
  1706→    totalCompanies: companies.length,
  1707→    totalCostUsd: totalUsd,
  1708→    providerDistribution: providerDist,
  1709→    modelDistribution: modelDist,
  1710→    totalTokenUsage: totalTokens,
  1711→    totalUsdSpent: totalUsd,
  1712→    avgCostPerRequest: totalUsd / totalRequests,
  1713→    avgCostPerInvoice: totalInvoices > 0 ? totalUsd / totalInvoices : 0,
  1714→    avgCostPerCompany: companies.length > 0 ? totalUsd / companies.length : 0,
  1715→    highestCostTenants,
  1716→    cacheHitRate,
  1717→    memoryHitRate,
  1718→    ruleHitRate,
  1719→    patternHitRate,
  1720→    learningImprovement: {
  1721→      firstHalfAvgCost: fhAvg,
  1722→      secondHalfAvgCost: shAvg,
  1723→      improvementPct,
  1724→    },
  1725→    requestsPerMinute: rqsPerMin,
  1726→    p50Latency: p50,
  1727→    p95Latency: p95,
  1728→    p99Latency: p99,
  1729→    errorRate,
  1730→    budgetBlockedCount: budgetBlocked,
  1731→  };

  1754→export function generateFounderReport(

  1756→  telemetry: TelemetryEntry[],
  1757→  metricsOrSeed?: MetricsSummary | number,
  1758→  options?: { seed?: number; companyCount?: number },

  1760→  const seed = typeof metricsOrSeed === 'number' ? metricsOrSeed : (options?.seed ?? 42);
  1761→  const metrics = typeof metricsOrSeed === 'object' && metricsOrSeed !== null ? metricsOrSeed as MetricsSummary : calculateMetrics(companies, telemetry);
  1762→  const totalInvoices = companies.reduce((s, c) => s + (c.invoices?.length ?? 0), 0);
  1763→  const totalProducts = companies.reduce((s, c) => s + (c.products?.length ?? 0), 0);
  1764→  const totalClients = companies.reduce((s, c) => s + (c.clients?.length ?? 0), 0);

  1789→    estimatedAwsCostMonthly.storage +

  1814→  const infrastructureBottlenecks: string[] = [];

  1832→  if (telemetry.filter(e => e.retries > 0).length / telemetry.length > 0.1) queueBottlenecks.push('Retry rate exceeds 10% — queue instability');
  1833→

  1835→  if (metrics.cacheHitRate < 0.35) aiBottlenecks.push('Low cache hit rate forces excessive AI calls');

  1839→  const endpointLatencies = new Map<string, { totalMs: number; count: number; maxMs: number; latencies: number[] }>();
